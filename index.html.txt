  <script>
    // THAY LINK NÀY bằng link affiliate/redirect của bạn
    const AFFILIATE_PREFIX = "https://s.shopee.vn/7VDAUWVMRG";

    function generateLink(){
      const input = document.getElementById("urlInput").value.trim();

      if(!input){
        alert("Vui lòng nhập link Shopee");
        return;
      }

      const encoded = encodeURIComponent(input);

      // Demo tạo link
      const finalLink =
        AFFILIATE_PREFIX +
        "?redirect=" +
        encoded +
        "&subid=myaff";

      document.getElementById("resultBox").style.display = "block";
      document.getElementById("resultLink").value = finalLink;
    }

    function copyLink(){
      const text = document.getElementById("resultLink");

      text.select();
      text.setSelectionRange(0, 99999);

      navigator.clipboard.writeText(text.value);

      alert("Đã copy link");
    }
  </script>
